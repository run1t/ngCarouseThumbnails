Examples

Basic

  • HTML
 <body ng-app="carouselApp" ng-controller="IndexController" >
   <carousel-thumbnails ng-model="images" > </carousel-thumbnails>
   <button ng-click="addImages()"> Add images </button>
 </body>
  • Javascript
<script>
        $angular.module('carouselApp', ['ngCarouselThumbnails'])
          .controller('IndexController', function IndexController($scope) {

            $scope.images = [
              'http://dummyimage.com/100x100/000/fff&text=1',
              'http://dummyimage.com/100x100/000/fff&text=2',
              'http://dummyimage.com/100x100/000/fff&text=3',
              'http://dummyimage.com/100x100/000/fff&text=4'
            ];
                      
            var images_iter = 4;
                      
            $scope.addImages = function(){
              $scope.images.push('http://dummyimage.com/100x100/000/fff&text='+(++images_iter));
            };
        });
</script>
